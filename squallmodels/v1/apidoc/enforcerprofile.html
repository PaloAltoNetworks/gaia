EnforcerProfile

Allows to create reusable configuration profile for your enforcers. Enforcer Profiles contains various startup information that can (for some) be updated live. Enforcer Profiles are assigned to some Enforcer using a Enforcer Profile Mapping Policy.

Retrieve all EnforcerProfiles in a given Enforcer

apoctl Go Python REST 
apoctl api list enforcerprofiles in enforcer parent-id -n /namespace

package main

import (
  "github.com/aporeto-inc/manipulate"
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  squallmodels "github.com/aporeto-inc/gaia/squallmodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects squallmodels.EnforcerProfilesList

  parent := squallmodels.NewEnforcer()
  parent.SetIdentifier("parent-id")

  mctx := manipulate.NewContext()
  mctx.Parent = parent

  if err = m.RetrieveMany(mctx, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from squallmodels import EnforcerProfile, Enforcer
from pymanipulate.context import Context
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = EnforcerProfile()
    objects = []

    parent = Enforcer(ID="parent-id")
    mctx = Context(parent=parent)

    try:
        manipulator.retrieve_children(mctx, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()
  
curl -X GET \
  -H "X-Namespace: namespace" \
  -H "Authorization: token" \
  http://<host>:<port>/enforcers/id/enforcerprofiles
